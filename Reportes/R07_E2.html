x = linspace(0,2*pi,16);           % creación del espacio
y = sin(x)+0.1*rand(1,length(x));  % creación de la función
Train_idx = 1:3:length(x);         % obtenemos nuestro espacio de Train dependiente del valor de x
Valid_idx = 2:3:length(x);         % obtenemos nuestra espacio de validación dependiente del valor de x
Test_idx = 3:3:length(x);          % obtenemos nuestro espacio de prueba dependiente del valor de x
figure, plot(x,y,'-k')
hold on
plot(x(Train_idx),y(Train_idx),'.b')
hold on, plot(x(Valid_idx),y(Valid_idx),'.g')
hold on
plot(x(Test_idx), y(Test_idx),'.r')
legend('model','train','validation','test')
TFi = {'tansig' 'purelin'};
BTF = 'trainlm';
BLF = 'learngdm';
PF = 'mse';
IPF = {'mapstd'};
OPF = {'mapstd'};
net = newff(x, y,[],TFi,BTF,BLF,PF,IPF,OPF);
net.divideFcn = 'divideind';
net.divideParam.trainInd = Train_idx;
net.divideParam.valInd = Valid_idx;
net.divideParam.testInd = Test_idx;
bas_stnet = init(net);               %Inicialización de la network
NN_conf = train(bas_stnet,x,y);      %Train
out.train = sim(NN_conf, x(:,Train_idx));   %Modelo de salida de Train
out.val = sim(NN_conf, x(:,Valid_idx));     %Modelo de salida de la validación
out.test = sim(NN_conf, x(:,Test_idx));     %Prueba
figure,                                     %Ploteado de las figuras
subplot(3,1,1),
plot(x(Train_idx),y(Train_idx),'.b'),
hold on,
plot(x(Train_idx),out.train,'.r'),
title('train'),
legend('target','neural net model')
subplot(3,1,2),
plot(x(Valid_idx),y(Valid_idx),'.b'),
hold on,
plot(x(Valid_idx),out.val,'.r'),
title('validation'),
legend('target','neural net model')
subplot(3,1,3),
plot(x(Test_idx), y(Test_idx),'.b'),
hold on,
plot(x(Test_idx),out.test,'.r'),
title('test'),
legend('target','neural net model')
