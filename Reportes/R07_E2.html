x = linspace(0,2*pi,16); % creaci贸n del espacio
y = sin(x) + 0.1*rand(1,length(x));% creaci贸n de la funci贸n
trainData = 1:5:length(x); % obtenemos nuestro espacio de Train dependiente del valor de x
validation = 2:5:length(x); % obtenemos nuestra espacio de validaci贸n dependiente del valor de x
testData = 3:5:length(x); % obtenemos nuestro espacio de prueba dependiente del valor de x
figure;
plot(x,y,'-k');
hold on;
plot(x(trainData),y(trainData),'.b');
plot(x(validation),y(validation),'.g');
plot(x(testData), y(testData),'.r');
legend('model','training','validation','test')
TFi = {'tansig','purelin'};
BTF = 'trainlm';
BLF = 'learngdm';
PF = 'mse';
IPF = {'mapstd'};
OPF = {'mapstd'};
net = newff(x, y,[],TFi,BTF,BLF,PF,IPF,OPF);
net.divideFcn = 'divideind';
net.divideParam.trainInd = trainData;
net.divideParam.valInd = validation;
net.divideParam.testInd = testData;
bas_stnet = init(net); %Inicializaci贸n de la network
NN_conf = train(bas_stnet,x,y); %Train
out.train = sim(NN_conf, x(:,trainData)); %Modelo de salida de Train
out.val = sim(NN_conf, x(:,validation)); %Modelo de salida de la validaci贸n
out.test = sim(NN_conf, x(:,testData)); %Prueba
figure; %Ploteado de las figuras
subplot(3,1,1);
plot(x(trainData),y(trainData),'.b');
hold on;
plot(x(trainData),out.train,'.r'), title('train');
legend('target','neural net model');
subplot(3,1,2);
plot(x(validation),y(validation),'.b'),
hold on;
plot(x(validation),out.val,'.r'), title('validation');
legend('target','neural net model');
subplot(3,1,3);
plot(x(testData), y(testData),'.b');
hold on;
plot(x(testData),out.test,'.r'), title('test');
