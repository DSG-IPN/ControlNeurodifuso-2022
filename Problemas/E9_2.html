d_F=[diff(F,x1);diff(F,x2)];
X(1,1)=0;
X(1,2)=0;
for i=1:100
eval=[X(i,1);X(i,2)];
g=[subs(d_F(1),[x1,x2],eval');subs(d_F(2),[x1,x2],eval')];
alpha=0.01;
prov=eval-(alpha*g);
X(i+1,1)=prov(1,1);
X(i+1,2)=prov(2,1);
end
figure
