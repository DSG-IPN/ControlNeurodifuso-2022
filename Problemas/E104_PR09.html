El limite de decisión final, se muestra perpendicular a los contornos y los puntos, que los divide cerca de la mitad, es decir, los separa lo mas cercano a la mitad de los puntos.
function [W,b] = adaline_epoca(W0,b0,alpha, p,t,ba)
% ba = 1 activa la actualizción del bias
if nargin > 5
ba = ba;
else
ba = 0;
end
[fil,col] = size(p);
W = [W0'];
b = [b0'];
for i = 1:col
a = W0 * p(:,i) + b0;
e = t(:,i)-a;
Wn = W0 + 2 * alpha * e * p(:,i)';
if ba
bn = b0 + 2 *alpha * e;
b = [b bn'];
b0 = bn;
end
W = [W Wn'];
W0 = Wn;
end
end
